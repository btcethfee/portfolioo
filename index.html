const canvas = document.createElement("canvas"); const ctx = canvas.getContext("2d"); document.body.appendChild(canvas); canvas.width = 800; canvas.height = 600; // Load spaceship image const spaceship = new Image(); spaceship.src = "https://upload.wikimedia.org/wikipedia/commons/3/3f/Space_shuttle.png"; // Player object const player = { x: canvas.width / 2 - 25, y: canvas.height - 80, width: 50, height: 50, speed: 5, bullets: [] }; // Asteroids array const asteroids = []; // Controls const keys = {}; document.addEventListener("keydown", (e) => keys[e.key] = true); document.addEventListener("keyup", (e) => keys[e.key] = false); document.addEventListener("keydown", (e) => { if (e.key === " ") { player.bullets.push({ x: player.x + 20, y: player.y, width: 10, height: 20 }); } }); // Game loop function update() { if (keys["ArrowLeft"] && player.x > 0) player.x -= player.speed; if (keys["ArrowRight"] && player.x < canvas.width - player.width) player.x += player.speed; player.bullets.forEach((bullet, index) => { bullet.y -= 7; if (bullet.y < 0) player.bullets.splice(index, 1); }); if (Math.random() < 0.02) { asteroids.push({ x: Math.random() * (canvas.width - 40), y: 0, width: 40, height: 40 }); } asteroids.forEach((asteroid, index) => { asteroid.y += 3; if (asteroid.y > canvas.height) asteroids.splice(index, 1); }); } function draw() { ctx.clearRect(0, 0, canvas.width, canvas.height); ctx.drawImage(spaceship, player.x, player.y, player.width, player.height); ctx.fillStyle = "red"; player.bullets.forEach(bullet => ctx.fillRect(bullet.x, bullet.y, bullet.width, bullet.height)); ctx.fillStyle = "gray"; asteroids.forEach(asteroid => ctx.fillRect(asteroid.x, asteroid.y, asteroid.width, asteroid.height)); } function gameLoop() { update(); draw(); requestAnimationFrame(gameLoop); } spaceship.onload = () => gameLoop();
